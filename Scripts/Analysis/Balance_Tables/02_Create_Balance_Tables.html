compute_summary_stats <- function(df) { df %>% mutate(variable = factor(variable, levels = unique(variable))) %>% # Preserve variable order group_by(group, variable) %>% summarise( mean_value = round(mean(value, na.rm = TRUE), 2), sd_value = round(sd(value, na.rm = TRUE), 2), N = sum(!is.na(value)), # Count of non-missing values .groups = “drop” ) %>% pivot_longer(cols = c(mean_value, sd_value, N), names_to = “stat_type”, values_to = “stat_value”) %>% mutate(stat_type = case_when( stat_type == “mean_value” ~ “Mean”, stat_type == “sd_value” ~ “SD”, stat_type == “N” ~ “N” )) %>% mutate(stat_type = factor(stat_type, levels = c(“Mean”, “SD”, “N”))) %>% # Set order of statistics arrange(variable, stat_type) %>% pivot_wider(names_from = group, values_from = stat_value) %>% select(variable, stat_type, everything()) # Ensure variable and stat_type come first }

Create function for regression coefficients

run_all_regressions <- function(data, treatment_col, village_col, vars_to_test) { results_list <- list()

for (var in vars_to_test) { tryCatch({ # Subset and drop rows with NA in outcome subset <- data %>% select(all_of(var), all_of(treatment_col), all_of(village_col)) %>% drop_na()

  # Define model
  formula <- as.formula(paste(var, "~", treatment_col))
  model <- lm(formula, data = subset)

  # Clustered variance-covariance
  clustered_vcov <- vcovCL(model, cluster = subset[[village_col]])
  clustered_se <- coeftest(model, vcov = clustered_vcov)

  # Wald test for F-statistic and p-value
  wald <- waldtest(model, vcov = clustered_vcov, test = "F")
  f_stat <- round(wald[2, "F"], 2)
  f_pval <- round(wald[2, "Pr(>F)"], 3)

  # Sample size used
  n_obs <- as.character(nobs(model))

  # Extract control group stats (intercept)
  control_mean <- as.character(round(clustered_se[1, 1], 2))
  control_se <- as.character(round(clustered_se[1, 2], 2))

  # Extract treatment comparisons
  coef_names <- rownames(clustered_se)[-1]
  stats <- as.character(round(clustered_se[-1, 1], 2))
  pvals <- as.character(round(clustered_se[-1, 4], 2))

  results <- data.frame(
    variable = var,
    comparison = coef_names,
    stat = stats,
    p = pvals,
    control_mean = control_mean,
    control_se = control_se,
    N = n_obs,
    f_statistic = as.character(f_stat),
    f_p_value = as.character(f_pval),
    row.names = NULL,
    stringsAsFactors = FALSE
  )

  results_list[[var]] <- results
}, error = function(e) {
  # In case of error, return NA for the variable
  results_list[[var]] <- data.frame(
    variable = var,
    comparison = NA,
    stat = NA,
    p = NA,
    control_mean = NA,
    control_se = NA,
    N = NA,
    f_statistic = NA,
    f_p_value = NA,
    row.names = NULL,
    stringsAsFactors = FALSE
  )
})

}

final_results <- bind_rows(results_list) return(final_results) }

compute_fdr_qvalues <- function(regression_results) {

# Apply Benjamini-Hochberg correction to regression p-values regression_results <- regression_results %>% mutate(q_value = as.character(round(p.adjust(p, method = “BH”), 2))) # Adjust and round q-values # Adjusting only regression p-values

return(regression_results) # Return the updated regression table with q-values }
